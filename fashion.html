function runEngine() { const input = { height: get("height"), bodyShape: get("bodyShape"), goal: get("goal"), legRatio: get("legRatio"), undertone: get("undertone"), contrast: get("contrast"), fit: get("fit"), occasion: get("occasion"), climate: get("climate"), style: get("style"), footwear: get("footwear") }; let attr = { silhouette: "balanced", palette: [], fabric: "", fit: input.fit || "regular", base: "casual" }; let explanation = []; let avoid = []; // 🔹 PROPORTION ENGINE (CORE) if (input.height === "short" || input.goal === "taller") { attr.silhouette = "vertical"; explanation.push("Vertical styling creates a taller appearance"); avoid.push("Avoid strong color breaks between top and bottom"); } if (input.legRatio === "short") { explanation.push("Higher waist pants help make legs look longer"); } if (input.goal === "slimmer") { attr.fit = "slim"; explanation.push("Slim fit reduces visual bulk"); avoid.push("Avoid oversized or baggy clothing"); } if (input.goal === "broader") { explanation.push("Layering and structure add visual width"); } // 🔹 BODY SHAPE BALANCING if (input.bodyShape === "triangle") { explanation.push("Lighter top balances heavier lower body"); avoid.push("Avoid tight or light-colored pants"); } if (input.bodyShape === "inverted") { explanation.push("Darker top reduces upper body dominance"); avoid.push("Avoid heavy jackets with shoulder padding"); } if (input.bodyShape === "oval") { attr.silhouette = "vertical"; explanation.push("Clean vertical lines reduce focus on midsection"); avoid.push("Avoid tight clothing around stomach"); } // 🔹 COLOR ENGINE if (input.undertone === "warm") { attr.palette = ["beige", "olive", "brown"]; explanation.push("Warm tones match your skin undertone"); } else { attr.palette = ["white", "grey", "black"]; explanation.push("Neutral tones create a clean and versatile look"); } if (input.contrast === "high") { explanation.push("High contrast outfits match your natural features"); } else { explanation.push("Soft color combinations create harmony"); } // 🔹 CLIMATE ENGINE if (input.climate === "hot") { attr.fabric = "light cotton or linen"; explanation.push("Breathable fabrics keep you cool"); avoid.push("Avoid heavy fabrics like thick denim or wool"); } else if (input.climate === "cold") { attr.fabric = "layered fabrics (jackets, wool)"; explanation.push("Layering helps retain warmth and structure"); } // 🔹 OCCASION ENGINE if (input.occasion === "formal") { attr.base = "formal"; } else if (input.occasion === "party") { attr.base = "party"; } else { attr.base = "casual"; } // 🔹 OUTFIT GENERATION (REALISTIC) let outfits = [ // SAFE { name: "Safe Option", top: `${attr.palette[0]} ${attr.base === "formal" ? "shirt" : "t-shirt"}`, bottom: "dark slim-fit trousers", shoes: input.footwear, fabric: attr.fabric, score: score(attr, input, "safe"), explanation: explanation.slice(0, 3), avoid: avoid.slice(0, 2) }, // BALANCED { name: "Balanced Option", top: `${attr.palette[1]} shirt`, bottom: "neutral chinos", shoes: input.footwear, fabric: attr.fabric, score: score(attr, input, "balanced"), explanation: explanation.slice(0, 4), avoid: avoid.slice(0, 3) }, // BOLD { name: "Bold Option", top: `${attr.palette[2]} statement shirt`, bottom: "contrast trousers", shoes: input.footwear, fabric: attr.fabric, score: score(attr, input, "bold"), explanation: explanation, avoid: avoid } ]; outfits.sort((a, b) => b.score - a.score); display(outfits); } // 🔹 SCORING (IMPROVED) function score(attr, input, type) { let score = 0; // Proportion (highest) if (attr.silhouette === "vertical") score += 15; // Context if (input.occasion) score += 10; if (input.climate) score += 10; // Fit if (input.fit === "slim") score += 10; // Style match if (input.style === "minimal" && type === "safe") score += 10; if (input.style === "street" && type === "bold") score += 10; // Color if (attr.palette.length) score += 5; return score; } function get(id) { return document.getElementById(id).value; } // 🔹 OUTPUT DISPLAY (UPGRADED) function display(outfits) { let html = "

Top Outfit Recommendations

"; outfits.forEach(o => { html += `

${o.name} (Score: ${o.score})

Top: ${o.top}

Bottom: ${o.bottom}

Shoes: ${o.shoes}

Fabric: ${o.fabric}

Why it works:

    ${o.explanation.map(e => `
  • ${e}
    • `).join("")}

Avoid:

    ${o.avoid.map(a => `
  • ${a}
    • `).join("")}
`; }); document.getElementById("result").innerHTML = html; }
